(declare-datatypes () ((Block a b c)))
(declare-fun OnTop (Block Block) Bool)
(declare-fun Green (Block) Bool)
(assert (OnTop a b))
(assert (OnTop b c))
(assert (Green a))
(assert (not (Green c)))
(declare-const x Block)
(declare-const y Block)
(assert (and (and (Green x) (not (Green y))) (OnTop x y)))
(check-sat)
